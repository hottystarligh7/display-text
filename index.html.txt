<!DOCTYPE html>
<html>
<head>
  <meta charset="UTF-8">
  <title>Text Display</title>
</head>
<body style="font-size:28px;padding:20px;">
  <textarea id="textArea" style="width:100%;height:120px;font-size:28px;"></textarea>
  <script>
    // URLパラメータからテキストを取得
    const params = new URLSearchParams(window.location.search);
    const text = params.get('text') || '';
    const textarea = document.getElementById('textArea');
    textarea.value = text;
    textarea.select(); // 自動選択してコピーしやすく
  </script>
</body>
</html>
